# 1. CUDA + cuDNN runtime base
FROM nvidia/cuda:12.9.1-cudnn-runtime-ubuntu22.04

# 2. Prevent interactive prompts
ENV DEBIAN_FRONTEND=noninteractive

# 3. Install system dependencies and Poetry
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
        python3 python3-pip python3-venv python3-dev \
        build-essential git curl wget && \
    rm -rf /var/lib/apt/lists/* && \
    python3 -m pip install --upgrade pip && \
    curl -sSL https://install.python-poetry.org | python3 -

# 4. Add poetry to PATH
ENV PATH="/root/.local/bin:$PATH"

# 5. Set working directory
WORKDIR /app

# 6. Copy poetry files and install dependencies
COPY pyproject.toml poetry.lock* /app/
RUN poetry install --no-root

# --- ADD THE FOLLOWING SECTION HERE (between dependencies and code copy) ---
# 6.1 Download ESM-1b weights (FAIR ESM checkpoint)
RUN mkdir -p /root/.cache/torch/hub/checkpoints && \
    wget -O /root/.cache/torch/hub/checkpoints/esm1b_t33_650M_UR50S.pt \
      https://dl.fbaipublicfiles.com/fair-esm/models/esm1b_t33_650M_UR50S.pt
# --- END SECTION ---

# 7. Copy the rest of your source code
COPY . /app

# 8. Expose the port you want your Flask app to serve (e.g., 6004)
EXPOSE 6004

# 9. Run Flask app with Waitress (make sure wsgi.py defines "app")
CMD ["poetry", "run", "waitress-serve", "--host=0.0.0.0", "--port=6004", "wsgi:app"]
